PREFIX owl: <http://www.w3.org/2002/07/owl#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX : <https://w3id.org/airco#>

SELECT ?archetype ?regulation WHERE {
  :FacialAttributeClassifier rdf:type ?archetype .
  FILTER(STRSTARTS(STR(?archetype), "https://w3id.org/airco#System_"))
  ?archetype rdfs:subClassOf ?restriction .
  ?restriction owl:onProperty :triggersRegulation ;
               owl:someValuesFrom ?regulation .
}
